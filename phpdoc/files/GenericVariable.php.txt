declare(strict_types=1);

namespace poggit\libasynql;

/**
 * Represents a variable that can be passed into {@link GenericStatement::format()}
 */
class GenericVariable{
	public $name;
	public $type;

	public function __construct(string $name, string $type){
		$this->name = $name;
		$this->type = $type;
	}


}
