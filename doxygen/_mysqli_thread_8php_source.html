72  protected function executeQuery(&$mysqli, int $mode, string $query, array $params) : SqlResult{
73  assert($mysqli instanceof mysqli);
74  if(empty($params)){
75  $result = $mysqli->query($query);
76  if($result === false){
77  throw new SqlError(SqlError::STAGE_EXECUTE, $mysqli->error, $query, []);
78  }
79  switch($mode){
80  case SqlThread::MODE_GENERIC:
81  case SqlThread::MODE_CHANGE:
82  case SqlThread::MODE_INSERT:
83  if($result instanceof mysqli_result){
84  $result->close();
85  }
86  if($mode === SqlThread::MODE_INSERT){
87  return new SqlInsertResult($mysqli->affected_rows, $mysqli->insert_id);
88  }
89  if($mode === SqlThread::MODE_CHANGE){
90  return new SqlChangeResult($mysqli->affected_rows);
91  }
92  return new SqlResult();
93 
94  case SqlThread::MODE_SELECT:
95  $ret = $this->toSelectResult($result);
96  $result->close();
97  return $ret;
98  }
99  }else{
100  $stmt = $mysqli->prepare($query);
101  if(!($stmt instanceof mysqli_stmt)){
102  throw new SqlError(SqlError::STAGE_PREPARE, $mysqli->error, $query, $params);
103  }
104  $types = implode(array_map(function($param) use ($query, $params){
105  if(is_string($param)){
106  return "s";
107  }
108  if(is_float($param)){
109  return "d";
110  }
111  if(is_int($param)){
112  return "i";
113  }
114  throw new SqlError(SqlError::STAGE_PREPARE, "Cannot bind value of type " . gettype($param), $query, $params);
115  }, $params));
116  $stmt->bind_param($types, ...$params);
117  if(!$stmt->execute()){
118  throw new SqlError(SqlError::STAGE_EXECUTE, $stmt->error, $query, $params);
119  }
120  switch($mode){
121  case SqlThread::MODE_GENERIC:
122  $ret = new SqlResult();
123  $stmt->close();
124  return $ret;
125 
126  case SqlThread::MODE_CHANGE:
127  $ret = new SqlChangeResult($stmt->affected_rows);
128  $stmt->close();
129  return $ret;
130 
131  case SqlThread::MODE_INSERT:
132  $ret = new SqlInsertResult($stmt->affected_rows, $stmt->insert_id);
133  $stmt->close();
134  return $ret;
135 
136  case SqlThread::MODE_SELECT:
137  $set = $stmt->get_result();
138  $ret = $this->toSelectResult($set);
139  $set->close();
140  return $ret;
141  }
142  }
143 
144  throw new InvalidArgumentException("Unknown mode $mode");
145  }

147  private function toSelectResult(mysqli_result $result) : SqlSelectResult{
148  $columns = [];
149  $columnFunc = [];
150 
151  while(($field = $result->fetch_field()) !== false){
152  if($field->length === 1){
153  if($field->type === MysqlTypes::TINY){
154  $type = SqlColumnInfo::TYPE_BOOL;
155  $columnFunc[$field->name] = function($tiny){
156  return $tiny > 0;
157  };
158  }elseif($field->type === MysqlTypes::BIT){
159  $type = SqlColumnInfo::TYPE_BOOL;
160  $columnFunc[$field->name] = function($bit){
161  return $bit === "\1";
162  };
163  }
164  }
165  if($field->type === MysqlTypes::LONGLONG){
166  $type = SqlColumnInfo::TYPE_INT;
167  $columnFunc[$field->name] = function($longLong) use ($field){
168  if($field->flags & MysqlFlags::UNSIGNED_FLAG){
169  if(bccomp($longLong, "9223372036854775807") === 1){
170  $longLong = bcsub($longLong, "18446744073709551616");
171  }
172  return (int) $longLong;
173  }
174 
175  return (int) $longLong;
176  };
177  }elseif($field->flags & MysqlFlags::TIMESTAMP_FLAG){
178  $type = SqlColumnInfo::TYPE_TIMESTAMP;
179  $columnFunc[$field->name] = function($stamp){
180  return strtotime($stamp);
181  };
182  }elseif($field->type === MysqlTypes::NULL){
183  $type = SqlColumnInfo::TYPE_NULL;
184  }elseif(in_array($field->type, [
185  MysqlTypes::VARCHAR,
186  MysqlTypes::STRING,
187  MysqlTypes::VAR_STRING,
188  ], true)){
189  $type = SqlColumnInfo::TYPE_STRING;
190  }
191  if(!isset($type)){
192  if($field->flags & MysqlFlags::NUM_FLAG){
193  $type = $field->decimal > 0 || $field->type === MysqlTypes::DECIMAL ? SqlColumnInfo::TYPE_FLOAT : SqlColumnInfo::TYPE_INT;
194  }else{
195  $type = SqlColumnInfo::TYPE_OTHER;
196  }
197  }
198  $columns[$field->name] = new MysqlColumnInfo($field->name, $type, $field->flags, $field->type);
199  }
200 
201  $rows = [];
202  while(($row = $result->fetch_assoc()) !== null){
203  foreach($row as $col => &$cell){
204  if(isset($columnFunc[$col])){
205  $cell = $columnFunc[$col]($cell);
206  }
207  }
208  unset($cell);
209  $rows[] = $row;
210  }
211 
212  return new SqlSelectResult($columns, $rows);
213  }
