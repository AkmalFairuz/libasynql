145  private function executeImpl(string $queryName, array $args, int $mode, callable $handler, ?callable $onError){
146  $queryId = $this->queryId++;
147  $trace = libasynql::isPackaged() ? null : new Exception("(This is the original stack trace for the following error)");
148  $this->handlers[$queryId] = function($result) use ($handler, $onError, $trace){
149  if($result instanceof SqlError){
150  $this->reportError($onError, $result, $trace);
151  }else{
152  try{
153  $handler($result);
154  }catch(Exception $e){
155  if(!libasynql::isPackaged()){
156  $prop = (new ReflectionClass(Exception::class))->getProperty("trace");
157  $prop->setAccessible(true);
158  $newTrace = $prop->getValue($e);
159  $oldTrace = $prop->getValue($trace);
160  for($i = count($newTrace) - 1, $j = count($oldTrace) - 1; $i >= 0 && $j >= 0 && $newTrace[$i] === $oldTrace[$j]; --$i, --$j){
161  array_pop($newTrace);
162  }
163  $prop->setValue($e, array_merge($newTrace, [
164  [
165  "function" => Terminal::$COLOR_YELLOW . "--- below is the original stack trace ---" . Terminal::$FORMAT_RESET,
166  ],
167  ], $oldTrace));
168  }
169  throw $e;
170  }catch(Error $e){
171  if(!libasynql::isPackaged()){
